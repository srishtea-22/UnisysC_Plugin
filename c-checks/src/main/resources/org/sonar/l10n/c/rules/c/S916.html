Why is this an issue?

Switch labels (`case` and `default`) must appear directly inside a `switch` statement. Nesting switch labels inside other blocks (like `if`, `for`, `while`, or compound blocks) violates C language rules and results in undefined behavior.

Switch labels are control flow constructs that work exclusively with `switch` statements. Placing them in nested blocks breaks the intended structure and can cause:

  • Compilation errors or warnings
  • Incorrect program flow
  • Undefined behavior
  • Difficulty in code maintenance and understanding

Example:

    int value = 5;
    
    // Noncompliant - case inside if block
    switch (value) {
        case 1:
            printf("One\n");
            break;
        default:
            if (value > 10) {
                case 2:        // ERROR - case nested in if
                    printf("Two\n");
                    break;
            }
    }
    
    // Noncompliant - case inside for loop
    switch (value) {
        case 1:
            for (int i = 0; i < 3; i++) {
                case 2:        // ERROR - case nested in for
                    printf("Two\n");
                    break;
            }
    }
    
    // Compliant - all cases directly in switch
    switch (value) {
        case 1:
            printf("One\n");
            break;
        case 2:
            printf("Two\n");
            break;
        default:
            printf("Other\n");
            break;
    }
    
    // Compliant - logic moved outside switch
    if (value > 10) {
        switch (value) {
            case 11:
                printf("Eleven\n");
                break;
            default:
                printf("Other\n");
                break;
        }
    }
